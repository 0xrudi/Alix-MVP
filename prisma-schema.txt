generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id                String   @id @default(uuid())
  walletAddress     String   @unique
  username          String
  bio               String?
  ensName           String?
  avatarUrl         String?
  additionalWallets String[]
  createdAt         DateTime @default(now())
  updatedAt         DateTime @updatedAt
}
